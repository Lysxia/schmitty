-- |The ⇾⁺-relation is a subrelation of < on depth.
⇾⁺-depth : l₁ ⇾⁺ l₃ → depth l₁ < depth l₃
⇾⁺-depth [ l₁⇾l₃ ] = ⇾-depth l₁⇾l₃
⇾⁺-depth (l₁⇾l₂ ∷ l₂⇾⁺l₃) = <-trans (⇾-depth l₁⇾l₂) (⇾⁺-depth l₂⇾⁺l₃)

private
  module WfViaDepth  = Wf.InverseImage {_<_ = _<_} depth
  module WfSubDepth₁ = Wf.Subrelation {_<₁_ = _⇾_} {_<₂_ = _<_ on depth} ⇾-depth
  module WfSubDepth₂ = Wf.Subrelation {_<₁_ = _⇾⁺_} {_<₂_ = _<_ on depth} ⇾⁺-depth

-- |The ⇾-relation is well-founded via depth.
⇾-wellFounded : WellFounded _⇾_
⇾-wellFounded = WfSubDepth₁.wellFounded (WfViaDepth.wellFounded WfNat.<-wellFounded)

-- |The ⇾⁺-relation is well-founded.
⇾⁺-wellFounded : WellFounded _⇾⁺_
⇾⁺-wellFounded = WfSubDepth₂.wellFounded (WfViaDepth.wellFounded WfNat.<-wellFounded)


--------------------------------------------------------
-- The sublogic relation ⇾⁺ is a strict partial order --
--------------------------------------------------------

-- |The ⇾-relation is irreflexive.
⇾-irrefl : Irreflexive _≡_ _⇾_
⇾-irrefl refl ()

-- |The ⇾-relation is asymmetric.
⇾-asym : Asymmetric _⇾_
⇾-asym l₁⇾l₂ l₂⇾l₁ = <-asym (⇾-depth l₁⇾l₂) (⇾-depth l₂⇾l₁)

-- |The ⇾-relation respects equality.
⇾-resp₂-≡ : _⇾_ Respects₂ _≡_
⇾-resp₂-≡ = (λ { refl p → p}) , (λ { refl p → p})

-- |The ⇾⁺-relation is irreflexive.
⇾⁺-irrefl : Irreflexive _≡_ _⇾⁺_
⇾⁺-irrefl l₁≡l₂ l₁⇾⁺l₂ = <-irrefl (cong depth l₁≡l₂) (⇾⁺-depth l₁⇾⁺l₂)

-- |The ⇾⁺-relation is asymmetric.
⇾⁺-asym : Asymmetric _⇾⁺_
⇾⁺-asym l₁⇾⁺l₂ l₂⇾⁺l₁ = <-asym (⇾⁺-depth l₁⇾⁺l₂) (⇾⁺-depth l₂⇾⁺l₁)

-- |The ⇾⁺-relation is transitive.
⇾⁺-trans : Transitive _⇾⁺_
⇾⁺-trans = Plus._++_

-- |The ⇾⁺-relation respects equality.
⇾⁺-resp₂-≡ : _⇾⁺_ Respects₂ _≡_
⇾⁺-resp₂-≡ = (λ { refl p → p}) , (λ { refl p → p})

-- |The ⇾⁺-relation is a strict partial order.
⇾⁺-isStrictPartialOrder : IsStrictPartialOrder _≡_ _⇾⁺_
⇾⁺-isStrictPartialOrder = record
  { isEquivalence = Eq.isEquivalence
  ; irrefl        = ⇾⁺-irrefl
  ; trans         = ⇾⁺-trans
  ; <-resp-≈      = ⇾⁺-resp₂-≡ }
